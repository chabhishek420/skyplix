       ALTER TABLE clicks ADD INDEX idx_is_bot is_bot TYPE minmax GRANULARITY 4;

       -- Add skip index on click_token for attribution lookups
       ALTER TABLE clicks ADD INDEX idx_click_token click_token TYPE bloom_filter(0.001) GRANULARITY 1;

       -- Conversions: add bloom filter on click_token for fast joins
       ALTER TABLE conversions ADD INDEX idx_conv_click_token click_token TYPE bloom_filter(0.001) GRANULARITY 1;

       -- Materialize indexes on existing data
       ALTER TABLE clicks MATERIALIZE INDEX idx_country_code;
       ALTER TABLE clicks MATERIALIZE INDEX idx_campaign_id;
       ALTER TABLE clicks MATERIALIZE INDEX idx_is_bot;
       ALTER TABLE clicks MATERIALIZE INDEX idx_click_token;
       ALTER TABLE conversions MATERIALIZE INDEX idx_conv_click_token;
