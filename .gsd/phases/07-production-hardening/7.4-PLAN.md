    1. Create `test/load/click_pipeline.js` (k6 script):
       ```javascript
       import http from 'k6/http';
       import { check, sleep } from 'k6';
       import { Rate, Trend } from 'k6/metrics';

       const errorRate = new Rate('errors');
       const clickLatency = new Trend('click_latency_ms', true);

       export const options = {
         scenarios: {
           sustained_load: {
             executor: 'constant-arrival-rate',
             rate: 1000,           // 1000 RPS
             timeUnit: '1s',
             duration: '60s',
             preAllocatedVUs: 100,
             maxVUs: 200,
           },
           spike_test: {
             executor: 'ramping-arrival-rate',
             startRate: 100,
             timeUnit: '1s',
             preAllocatedVUs: 50,
             maxVUs: 500,
             stages: [
               { duration: '30s', target: 500 },
               { duration: '30s', target: 2000 },
               { duration: '30s', target: 5000 },
               { duration: '30s', target: 100 },
             ],
             startTime: '70s',
           },
         },
         thresholds: {
           'http_req_duration{scenario:sustained_load}': ['p(99)<5'],   // <5ms p99
           'http_req_duration{scenario:spike_test}': ['p(99)<10'],      // <10ms p99 under spike
           'errors': ['rate<0.01'],                                      // <1% error rate
         },
       };

       const BASE_URL = __ENV.BASE_URL || 'http://localhost:8080';

       export default function () {
         // Simulate L1 click with randomized params
         const campaignAliases = ['test-campaign', 'default'];
         const alias = campaignAliases[Math.floor(Math.random() * campaignAliases.length)];

         const res = http.get(`${BASE_URL}/${alias}?sub_id_1=k6test&sub_id_2=${__VU}`, {
           redirects: 0,  // Don't follow redirects — we're testing pipeline latency
           tags: { name: 'L1_click' },
         });

         clickLatency.add(res.timings.duration);

         check(res, {
           'status is 302 or 200': (r) => r.status === 302 || r.status === 200 || r.status === 404,
           'latency < 10ms': (r) => r.timings.duration < 10,
         });

         errorRate.add(res.status >= 500);
       }
       ```
    2. Create `test/load/README.md` with instructions:
       - Prerequisites: `brew install k6` or `go install go.k6.io/k6@latest`
       - Running: `k6 run test/load/click_pipeline.js`
       - With custom base URL: `k6 run -e BASE_URL=http://prod:8080 test/load/click_pipeline.js`
       - Interpreting results: p99 must be <5ms for PASS
    3. The script must NOT require any specific campaign data — it should work against a fresh instance where unknown aliases return 404 (testing pipeline overhead, not routing logic).
  </action>
